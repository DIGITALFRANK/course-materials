\newcommand{\headerq}{\header{Linear Algebra}{\#3}}

\newcommand{\quiz}[1]{
\namefield

\headerq

\instcalc{55}{65}{40}

\prob{1} (10+5+10+5 pts) Consider the system $A \vec{x} = b$ given by  
% 3 x 5 system
#1
\begin{itemize}
\item[(a) ] What is the reduced row echelon form of the coefficient matrix $A$? 
\item[(b) ] What are the pivot variables? What are the free variables?
\item[(c) ] What is the solution $\vec{x}$ to the system $A \vec{x} = b$? Give in terms of the special solution(s). 
\item[(d) ] What is the dimension of the solution set? What geometric shape does the solution have, inside what space? 
\end{itemize}

\pagebreak

\headerq

\prob{2} (10+5+10+5+5 pts) Consider the system $A^t \vec{y} = 0$ given by $A$ in Problem 1. 
% 5 x 3 system
\begin{itemize}
\item[(a) ] What is the reduced row echelon form of the coefficient matrix $A^t$? 
\item[(b) ] What are the pivot variables? What are the free variables?
\item[(c) ] What is the solution $\vec{y}$ to the system $A^t \vec{y} = 0$? Give in terms of the special solution(s). 
\item[(d) ] What is the dimension of the solution set? What geometric shape does the solution have, inside what space? 
\item[(e) ] What is $b \cdot \vec{y}$? What does that hint to you about the spaces in which these vectors exist?
\end{itemize}
}
